// https://docs.microsoft.com/en-us/powerapps/maker/canvas-apps/functions/function-split

// single column table with "Result" as col name
>> Split("this is a test", " ")
Table({Value:"this"},{Value:"is"},{Value:"a"},{Value:"test"})

// no delimeter returns each char
>> Split("test", "")
Table({Value:"t"},{Value:"e"},{Value:"s"},{Value:"t"})

// no delimeter returns each char
>> Split("Hello world", If(1 < 0, ""))
Table({Value:"H"},{Value:"e"},{Value:"l"},{Value:"l"},{Value:"o"},{Value:" "},{Value:"w"},{Value:"o"},{Value:"r"},{Value:"l"},{Value:"d"})

// two delimeters in a row in source text produces empty string
>> Split( "Hello, World", "l" )
Table({Value:"He"},{Value:""},{Value:"o, Wor"},{Value:"d"})

// string as delimeter
>> Split( "Hello, World", "ll" )
Table({Value:"He"},{Value:"o, World"})

// delimeter doesn't exist
>> Split( "Hello, World", "%" )
Table({Value:"Hello, World"})

// only delimeter is removed
>> Split( "Hello, World", "," )
Table({Value:"Hello"},{Value:" World"})

>> Split("", "")
Table()

>> Split(If(1 < 0, ""), "")
Table()

>> Split(Text(1/0), "")
Error({Kind:ErrorKind.Div0})
